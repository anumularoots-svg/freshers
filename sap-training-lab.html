# Tag each VM with student number

  aws ec2 create-tags \
    --resources $(aws ec2 describe-instances \
      --filters "Name=tag:Name,Values=SAP-Student-VM" \
      --query "Reservations[$((i-1))].Instances[0].InstanceId" \
      --output text) \
    --tags "Key=StudentVM,Value=Student-$i"
done

Setup Auto Start/Stop (EventBridge + Lambda)

Scheduled automation configure cheyandi cost savings kosam

# Lambda Function - Start VMs
import boto3
import json

def lambda_handler(event, context):
    ec2 = boto3.client('ec2')
    
    # Get all training instances
    response = ec2.describe_instances(
        Filters=[
            {'Name': 'tag:Environment', 'Values': ['Training']},
            {'Name': 'instance-state-name', 'Values': ['stopped']}
        ]
    )
    
    instance_ids = []
    for reservation in response['Reservations']:
        for instance in reservation['Instances']:
            instance_ids.append(instance['InstanceId'])
    
    if instance_ids:
        ec2.start_instances(InstanceIds=instance_ids)
        print(f'Started {len(instance_ids)} instances')
        return {'statusCode': 200, 'body': f'Started {len(instance_ids)} VMs'}
    
    return {'statusCode': 200, 'body': 'No instances to start'}

#!/bin/bash
# sap-lab-manager.sh - SAP Training Lab Management

case $1 in
  start)
    echo "üöÄ Starting all SAP Training VMs..."
    aws ec2 start-instances \
      --instance-ids $(aws ec2 describe-instances \
        --filters "Name=tag:Environment,Values=Training" \
        --query 'Reservations[].Instances[].InstanceId' \
        --output text)
    ;;
  
  stop)
    echo "üõë Stopping all SAP Training VMs..."
    aws ec2 stop-instances \
      --instance-ids $(aws ec2 describe-instances \
        --filters "Name=tag:Environment,Values=Training" \
        --query 'Reservations[].Instances[].InstanceId' \
        --output text)
    ;;
  
  status)
    echo "üìä VM Status:"
    aws ec2 describe-instances \
      --filters "Name=tag:Environment,Values=Training" \
      --query 'Reservations[].Instances[].[Tags[?Key==`StudentVM`].Value|[0],State.Name,PrivateIpAddress,PublicIpAddress]' \
      --output table
    ;;
  
  reset)
    echo "üîÑ Resetting VMs to clean snapshot..."
    # Stop, restore snapshot, start
    ./sap-lab-manager.sh stop
    sleep 60
    # Add snapshot restore logic here
    ./sap-lab-manager.sh start
    ;;
  
  *)
    echo "Usage: $0 {start|stop|status|reset}"
    ;;
esac
